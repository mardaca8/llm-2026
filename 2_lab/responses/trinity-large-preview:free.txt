```python
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Load the CSV file
df = pd.read_csv('sales.csv')

# Ensure date column is datetime format and sort by date
df['date'] = pd.to_datetime(df['date'])
df = df.sort_values('date')

# Calculate 30-day rolling average of revenue
df['rolling_avg'] = df['revenue'].rolling(window=30, min_periods=1).mean()

# Calculate outliers beyond 2 standard deviations
mean_revenue = df['revenue'].mean()
std_revenue = df['revenue'].std()
df['outlier'] = (np.abs(df['revenue'] - mean_revenue) > 2 * std_revenue)

# Create the plot
plt.figure(figsize=(14, 8))

# Plot raw revenue data
plt.plot(df['date'], df['revenue'], label='Revenue', alpha=0.5, color='#1f77b4')

# Plot rolling average
plt.plot(df['date'], df['rolling_avg'], label='30-Day Rolling Average', 
         color='#ff7f0e', linewidth=2)

# Highlight outliers
outliers = df[df['outlier']]
plt.scatter(outliers['date'], outliers['revenue'], color='red', 
            label='Outliers (>2 std)', s=80, zorder=5, edgecolors='black')

# Formatting
plt.xlabel('Date')
plt.ylabel('Revenue')
plt.title('Revenue with 30-Day Rolling Average (Outliers in Red)')
plt.legend()
plt.grid(True, alpha=0.3)
plt.tight_layout()

# Show plot
plt.show()
```
