#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
sales_analysis.py
-----------------
Loads a CSV file named 'sales.csv', computes a 30‑day rolling average of the'revenue' column, detects outliers (> 2 σ from the mean), and visualises:

    • Raw revenue (blue line)
    • 30‑day rolling average (orange line)
    • Outliers (red scatter points)

Requirements
------------
pandas, matplotlib"""

import pandas as pd
import matplotlib.pyplot as plt

def main():
    # -------------------------------------------------    # 1. Load the CSV
    # -------------------------------------------------
    # Assuming the CSV has at least two columns:
    #   - a date column (named 'date' or parseable as datetime)
    #   - a numeric revenue column named 'revenue'
    # Adjust the column names if your file differs.
    try:
        df = pd.read_csv('sales.csv', parse_dates=['date'])
    except Exception as e:
        raise FileNotFoundError(
            "Unable to read 'sales.csv'. Make sure the file exists and "
            "contains a parsable 'date' column."
        ) from e    # Ensure the data is sorted chronologically – required for a correct rolling window
    df = df.sort_values('date').reset_index(drop=True)

    # -------------------------------------------------
    # 2. 30‑day rolling average of revenue
    # -------------------------------------------------
    # Using a closed window that includes the current observation.
    # min_periods=1 lets the first 29 days still produce a value (based on available data).
    df['rolling_30d'] = df['revenue'].rolling(window=30, min_periods=1, closed='both').mean()

    # -------------------------------------------------
    # 3. Detect outliers (> 2 std deviations from the mean)
    # -------------------------------------------------    rev_mean = df['revenue'].mean()
    rev_std  = df['revenue'].std()

    # Boolean mask: True where the point is an outlier
    outlier_mask = (df['revenue'] < rev_mean - 2 * rev_std) | (df['revenue'] > rev_mean + 2 * rev_std)
    df['is_outlier'] = outlier_mask

    # -------------------------------------------------    # 4. Plot raw data, rolling average, and outliers
    # -------------------------------------------------    plt.figure(figsize=(12, 6))

    # Raw revenue (light blue line)
    plt.plot(df['date'], df['revenue'],
             label='Revenue (raw)', color='steelblue', linewidth=1)

    # 30‑day rolling average (orange line)
    plt.plot(df['date'], df['rolling_30d'],
             label='30‑day Rolling Avg', color='orange', linewidth=2)

    # Highlight outliers in red (scatter on top of the raw line)
    plt.scatter(df.loc[outlier_mask, 'date'],
                df.loc[outlier_mask, 'revenue'],
                color='red',
                zorder=5,
                label='Outlier (>2σ)',
                edgecolor='k',
                s=60)

    # Beautify the plot
    plt.title('Daily Revenue with 30‑Day Rolling Average & Outliers')
    plt.xlabel('Date')
    plt.ylabel('Revenue')
    plt.legend()
    plt.grid(True, which='both', linestyle='--', linewidth=0.5, alpha=0.7)
    plt.tight_layout()
    plt.show()


if __name__ == '__main__':
    main()

### How it works
| Step | What the code does |
|------|--------------------|
| **Load CSV** | `pd.read_csv(..., parse_dates=['date'])` reads the file and converts the `date` column to `datetime`. |
| **Sort** | Guarantees chronological order for a correct rolling calculation. |
| **Rolling average** | `df['revenue'].rolling(window=30, min_periods=1).mean()` yields a 30‑day moving mean (first 29 days use whatever data is available). |
| **Outlier detection** | Computes global mean & std of `revenue`; flags points where `revenue < μ‑2σ` or `revenue > μ+2σ`. |
| **Plotting** | Raw revenue (blue line), rolling average (orange line), outliers (red scatter points). A legend, grid, and labels make the figure easy to interpret. |

#### Customisation tips
- **Different date column:** change `parse_dates=['date']` and the subsequent references (`df['date']`) to match your column name.
- **Adjust outlier threshold:** replace `2` in the outlier condition with any other multiplier (e.g., `3` for a stricter rule).
- **Window size:** modify `window=30` to another integer if you need a different look‑back period.
- **Saving the figure:** add `plt.savefig('sales_analysis.png', dpi=300)` before `plt.show()` if you want to store the plot.
